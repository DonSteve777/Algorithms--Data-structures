   void rotaIzq(Link & r1) {
      Link r2 = r1->dr;
      r1->dr = r2->iz;	OJO a esto !!!!
      r2->iz = r1;
      r1->altura = std::max(altura(r1->iz), altura(r1->dr)) + 1;
      r2->altura = std::max(altura(r2->iz), altura(r2->dr)) + 1;
      r1 = r2;
   }
